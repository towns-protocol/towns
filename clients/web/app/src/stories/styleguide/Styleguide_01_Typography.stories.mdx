## Colors

<Box>
  {Object.keys(vars.color.text).map((c) => (
    <Row
      key={c}
      label={c}
      backgroundVariant={
        c.match(/inverted/i)
          ? "inverted"
          : c.match(/onTone/)
          ? "accent"
          : undefined
      }
    >
      <Paragraph color={c}>{c}</Paragraph>
    </Row>
  ))}
</Box>
